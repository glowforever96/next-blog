**3. selectors, media query**

#### Selectors

```tsx
// button.css.ts
import { style } from "@vanilla-extract/css";
import { vars } from "@/styles/globals.css";

export const button = style({
  padding: 10,
  color: vars.colors.primary[50],
  fontSize: 20,

  // selectors
  ":hover": {
    backgroundColor: "red",
  },
  // or
  selectors: {
    "&:active": {
      color: "blue",
    },
    "nav li > &": {
      color: "green",
    },
  },
});
```
